Novadesk provides several timer functions to schedule code execution, similar to those found in browsers and Node.js.

<Admonition type="warning">
Timer functions are **only available in the Main script**. UI scripts should use main script for timing logic and communicate via [IPC](/api/widget-api/widget-methods/#inter-process-communication-ipc).
</Admonition>

<Method
  signature="setTimeout(callback, delay)"
  description="Execute a function after a specified delay."
>
### Parameters
  <MethodParam name="callback" type="function">
    Function to execute after the specified delay.
  </MethodParam>
  <MethodParam name="delay" type="number">
    Delay in milliseconds before the callback is executed.
  </MethodParam>
  ### Return
  <ReturnValue type="number">
    Timer ID that can be used with `clearTimeout`.
  </ReturnValue>
  ## Examples
### Execute a Function After Delay

<CodeBlock language="js">
  {`// Execute a function after 3 seconds
var timerId = setTimeout(function() {
  console.log("This runs after 3 seconds");
}, 3000);

// Cancel the timer before it executes
// clearTimeout(timerId);`}
</CodeBlock>
</Method>

{/* ========================================================================== */}

<Method
  signature="setInterval(callback, interval)"
  description="Execute a function repeatedly at specified intervals."
>
  ### Parameters

  <MethodParam name="callback" type="function">
    Function to execute repeatedly.
  </MethodParam>
  <MethodParam name="interval" type="number">
    Interval in milliseconds between executions.
  </MethodParam>
  ### Return
  <ReturnValue type="number">
    Timer ID that can be used with `clearInterval`.
  </ReturnValue>
  ## Examples
  ### Execute a Function Repeatedly

<CodeBlock language="js">
  {`// Execute a function every 2 seconds
var intervalId = setInterval(function() {
  console.log("This runs every 2 seconds");
}, 2000);

// Cancel the interval after 10 seconds
setTimeout(function() {
  clearInterval(intervalId);
  console.log("Interval stopped");
}, 10000);`}
</CodeBlock>
</Method>

{/* ========================================================================== */}

<Method signature="clearTimeout(id)"
  description="Cancel a timer created with setTimeout.">
  ### Parameters
  <MethodParam name="id" type="number">
    Timer ID returned by `setTimeout`.
  </MethodParam>
  ### Examples
  ### Cancel a Timeout

<CodeBlock language="js">
  {`var timerId = setTimeout(function() {
  console.log("This will not run");
}, 5000);

// Cancel the timer
clearTimeout(timerId);
console.log("Timer cancelled");`}
</CodeBlock>
</Method>

{/* ========================================================================== */}

<Method signature="clearInterval(id)"
  description="Cancel a timer created with setInterval.">
  ### Parameters
  <MethodParam name="id" type="number">
    Timer ID returned by `setInterval`.
  </MethodParam>
  ### Examples
<CodeBlock language="js">
  {`var intervalId = setInterval(function() {
  console.log("This runs repeatedly");
}, 1000);

// Stop the interval after 5 seconds
setTimeout(function() {
  clearInterval(intervalId);
  console.log("Interval stopped");
}, 5000);`}
</CodeBlock>
</Method>

{/* ========================================================================== */}

<Method
  signature="setImmediate(callback)"
  description="Execute a function as soon as possible in the next event loop iteration."
>
  ### Parameters
  <MethodParam name="callback" type="function">
    Function to execute in the next event loop iteration.
  </MethodParam>
  ### Return
  <ReturnValue type="number">
    ID that can be used to cancel the immediate callback.
  </ReturnValue>
  ### Examples
  ### Execute in Next Event Loop

<CodeBlock language="js">
  {`console.log("First");

setImmediate(function() {
  console.log("This runs after current execution completes");
});

console.log("Second");

// Output order:
// First
// Second
// This runs after current execution completes`}
</CodeBlock>
</Method>
